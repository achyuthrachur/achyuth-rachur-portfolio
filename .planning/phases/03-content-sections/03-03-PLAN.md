From src/components/reactbits/AnimatedList.tsx:
```typescript
interface AnimatedListProps {
  children: ReactNode;
  className?: string;
  itemClassName?: string;  // NOT used for badge classes — pass badge classes on span elements
  delay?: number;          // Default: 80 (milliseconds between items)
  duration?: number;       // Default: 0.5 (seconds per item animation)
  threshold?: number;      // Default: 0.1
  rootMargin?: string;     // Default: '-50px'
  from?: { opacity?: number; y?: number; x?: number; scale?: number };
  to?: { opacity?: number; y?: number; x?: number; scale?: number };
}
```

AnimatedList behavior (confirmed from source read):
- Wraps each child in `<motion.div variants={itemVariants}>` — motion.divs are block-level
- Uses `useInView(ref, { once: true })` — fires once when container enters viewport
- Uses `[0.16, 1, 0.3, 1]` easing internally (Crowe ease-out)
- Default from: `{ opacity: 0, y: 20 }`, to: `{ opacity: 1, y: 0 }`
- `className` prop applies to the outer `motion.div` container (NOT to individual items)

CRITICAL: AnimatedList wraps children in block-level motion.divs.
Use `className="flex flex-col gap-2"` on AnimatedList (vertical column layout).
Do NOT use `flex flex-wrap` — motion.divs are block-level and won't wrap inline.
Badge classes go on the `<span>` elements, not on AnimatedList.

Badge specification (custom spans — NOT shadcn Badge):
```tsx
<span className="bg-white text-[#2d3142] border-none shadow-crowe-sm rounded-full px-3 py-1.5 text-sm font-medium cursor-default transition-all duration-200 hover:shadow-amber-glow hover:scale-[1.03] inline-block">
  skill text
</span>
```
Note: `inline-block` on the span ensures it renders correctly within the block-level motion.div wrapper.
Note: `py-1.5` (6px) instead of `py-1` (4px) for slightly more comfortable pill height.

```tsx
    ],

  {
    title: 'Risk, Compliance & Governance',

      'Risk & compliance process automation',
      'Control design, evidence, and defensible documentation',
      'Stakeholder management (risk/compliance/technology alignment)',

  },
  {
    title: 'Model Risk Management',
    skills: [

      'Model performance & stability assessment',

      'CECL familiarity (WARM, DCF) and quantitative testing methods (e.g., OLS; HAC-adjusted techniques)',
    ],
  },
  {
    title: 'Domain Use Cases',
    skills: [
      'Audit analytics & automation',
      'Financial crime / AML operations support (triage, investigation enablement concepts)',
      'Model validation automation & reviewer support',
      'Reconciliation automation & exception management',
    ],
  },
];

interface SkillGroupProps {
  title: string;
  skills: string[];
}

function SkillGroup({ title, skills }: SkillGroupProps) {
  return (
    <div>
      <h3 className="text-lg font-semibold text-crowe-indigo-dark font-body">{title}</h3>
      <div className="h-0.5 w-8 bg-crowe-amber mt-1 mb-4" />
      <AnimatedList className="flex flex-col gap-2" delay={80} duration={0.5}>
        {skills.map((skill) => (
          <span
            key={skill}
            className="bg-white text-[#2d3142] border-none shadow-crowe-sm rounded-full px-3 py-1.5 text-sm font-medium cursor-default transition-all duration-200 hover:shadow-amber-glow hover:scale-[1.03] inline-block"
          >
            {skill}
          </span>
        ))}
      </AnimatedList>
    </div>
  );
}

export function SkillsSection() {
  return (
    <div className="max-w-5xl mx-auto px-6 py-24">
      <h2 className="text-3xl font-semibold text-crowe-indigo-dark font-body mb-2">Skills</h2>
      <div className="h-0.5 w-12 bg-crowe-amber rounded-full mb-10" />
      <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
        {SKILL_GROUPS.map((group) => (
          <SkillGroup key={group.title} title={group.title} skills={group.skills} />
        ))}
      </div>
    </div>
  );
}
```
